static NonVar stringToChars(String s) {
    if (0 == s.length())
        return Constant.aNil;
    Cons l = new Cons(new Int((s.charAt(0))), Constant.aNil);
    Cons curr = l;
    for (int i = 1; i < s.length(); i++) {
        Cons tail = new Cons(new Int((s.charAt(i))), Constant.aNil);
        curr.args[1] = tail;
        curr = tail;
    }
    return l;
}
